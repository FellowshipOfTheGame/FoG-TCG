return{
	title = "Tritao",
	type = 'c',
	aspects = {0, 0, 0, 0},
	desc = "Ganha +1/+2 para cada aspecto de Agua que voce controlar",
	flavor = "Nas profundezas dos Grandes Lagos, uma nacao de guerreiros vive em isolamento, especialistas em tecnicas marciais com o tridente.",
	cost = 4,
	atk = 1,
	hp = 2,

	OnEnter = function(args)
		local self = args[2]
		local board = args[1]
		local player = board.players[board.currPlayer]
		self.Data.hp = self.Data.hp + (2 * player.aspects[2])
		self.Data.atk = self.Data.atk + 1
		print(self.Data.hp)
	end
}
