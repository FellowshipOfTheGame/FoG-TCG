return{
	title = "Oceano",
	type = 't',
	aspects = {0, 0, 0, 0},
	elements = {0, 1, 0, 0},
	desc = "Gera 1 aspecto de Agua",
	flavor = "Apesar da maioria nao conseguir respirar embaixo de agua, os mares de Caldyr sao otimos meios de transporte, e fontes dos mais exoticos materiais.",
	cost = 1,

	OnEnter = function(args)
		local board = args[1];
		generate = {0, 1, 0, 0};
		board.AddAspects(board.currPlayer, generate);
	end,

	OnExit = function(args)
		local board = args[1];
		generate = {0,-1,0,0};
		board.AddAspects(3-board.currPlayer, generate);
	end
}
