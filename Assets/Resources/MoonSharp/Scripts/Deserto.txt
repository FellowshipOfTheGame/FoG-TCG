return{
	title = "Deserto",
	type = 't',
	aspects = {0, 0, 0, 0},
	elements = {1, 0, 0, 0},
	desc = "Gera 1 aspecto de Fogo",
	flavor = "Seu aspecto desolado esconde um complexo ecosistema debaixo de suas dunas, pronto para treinar os mais formidaveis guerreiros.",
	cost = 1,
	
	OnEnter = function(args)
		local board = args[1];
		generate = {1, 0, 0, 0};
		board.AddAspects(board.currPlayer, generate);
	end,


	OnExit = function(args)
		local board = args[1];
		generate = {-1,0,0,0};
		board.AddAspects(3-board.currPlayer, generate);
	end
}
