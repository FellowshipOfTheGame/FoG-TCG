return{
	title = "Pantano",
	type = 't',
	aspects = {0, 0, 0, 0},
	elements = {0, 1, 1, 0},
	desc = "Gera 1 aspecto de Agua e 1 aspecto de Terra",
	flavor = "Na superficie, e o local de descanso das mais variadas criaturas. Uma inspecao mais cautelosa revela a quantidade de criacao e renovacao ocorrendo ali mesmo, dentre as aguas lamacentas e as arvores retorcidas.",
	cost = 3,
	
	OnEnter = function(args)
		local board = args[1];
		generate = {0, 1, 1, 0};
		board.AddAspects(board.currPlayer, generate);
	end,

	OnExit = function(args)
		local board = args[1];
		generate = {0,-1,-1,0};
		board.AddAspects(3-board.currPlayer, generate);
	end
	}
