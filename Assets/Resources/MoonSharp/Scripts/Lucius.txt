return{
	title = "Lucius",
	type = 'c',
	aspects = {0, 1, 1, 0},
	desc = "Envenenar",
	flavor = "",
	cost = 3,
	atk = 2,
	hp = 5,

	OnDamageDealt = function(args)
		local self = args[2]
		local enemy = args[3]
		if enemy ~= nil then
			enemy.addModifier(self, "Poisoned");
		end
	end,

	Poisoned = function(args)
		local self = args[2]
		self.addAnimation("hit");
		self.Data.hp = self.Data.hp - 1
		if self.Data.hp <= 0 then
			self.Die();
		end
	end
}
