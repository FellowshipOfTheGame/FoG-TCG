return{
	title = "Sol Forte",
	type = 'a',
	aspects = {0, 0, 0, 0},
	desc = "Todas as criaturas com aspecto e fogo tem +1/+1",
	flavor = "Em dias como esse, pessoas matariam por um pouco de chuva.",
	cost = 0,

	OnEnter = function(args)
		local board = args[1]
		local self = args[2]
		for n=1,2 do
			for i=0,4 do
				local card = board.GetMatrixValue(n,i)
				if card ~= nil then
					if card.Data.aspects[1] > 0 then
						card.changeAnimation("buff");
						card.Data.atk = card.Data.atk + 1;
					end
					print(card.Data.atk)
				end
			end
		end
	end,

	OnNewCardInField = function(args)
		local card = args[3]
			if card.type == 'c' and card.Data.aspects[1] > 0 then
				card.changeAnimation("buff");
				card.Data.atk = card.Data.atk + 1;
			end
	end,

	OnExit = function(args)
		local board = args[1]
		local self = args[2]
		for n=1,2 do
			for i=0,4 do
				local card = board.GetMatrixValue(n,i)
				if card ~= nil then
					if card.Data.aspects[1] > 0 then
						card.changeAnimation("debuff");
						card.Data.atk = card.Data.atk - 1;
					end
				end
				print(self.Data.atk)
			end
		end
	end
}
