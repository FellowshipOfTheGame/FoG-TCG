return{
	title = "Faust",
	type = 'c',
	aspects = {0, 0, 1, 0},
	desc = "Reposicionar, Recebe menos 1 de dano em ataques",
	flavor = "Faust treinou toda sua vida em uma ordem reclusa de monges. Depois de anos de disciplina e trabalho duro, tornou-se um guerreiro extremamente formidável.",
	cost = 3,
	atk = 2,
	hp = 4

	abilities = {

		reposition = function(self, board)
		local pos = board.requestSelectPosition(alliedPosition)
			if pos then
				board.swap(self.position, pos)
			end
		end		
		}

	onIncomingDamage = function(self, attacker, args)
		args[1] = args[1] - 1;
	end
	}
