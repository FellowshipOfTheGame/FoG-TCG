return{
	title = "Rio",
	type = 't',
	aspects = {0, 0, 0, 0},
	elements = {0, 1, 0, 0},
	desc = "Gera 1 aspecto de Agua",
	flavor = "Rios sao a manifestacao do ciclo infinito da mudanca: voce nunca tocara nessa mesma agua, nesse mesmo trecho de rio novamente.",
	cost = 1,

	OnEnter = function(args)
		local board = args[1];
		generate = {0, 1, 0, 0};
		board.AddAspects(board.currPlayer, generate);
	end,

	OnExit = function(args)
		local board = args[1];
		generate = {0,-1,0,0};
		board.AddAspects(3-board.currPlayer, generate);
	end
}
