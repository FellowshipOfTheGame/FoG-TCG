return{
	title = "Volgarr",
	type = 'c',
	aspects = {0, 0, 0, 0},
	desc = "Atropelar",
	flavor = "No caminho desse guerreiro ha apenas a morte. Nao a dele, e claro.",
	cost = 4,
	atk = 3,
	hp = 6,

	OnDamageDealt = function(args)
		local board, self, enemy, dmg = unpack(args)
		local player = 6/(board.currPlayer * 3)
		if self.Data.atk > enemy.Data.hp then
			board.players[player].hP = board.players[player].hP + enemy.Data.hp
		end
	end
	}
