return{
	title = "Hikari",
	type = 'c',
	aspects = {0, 1, 0, 0},
	desc = "Pague 1 mana, Hikari ganha +1/+0 nesse turno",
	flavor = "Com o terrível espírito do lobo de nove caudas preso em seu corpo, Hikari procura balancear sua humanidade e bestialidade enquanto viaja pelo mundo, tentando usar seus poderes para corrigir o mal aonde quer que ele esteja.",
	cost = 2,
	atk = 2,
	hp = 3,

	buff = 0,

	OnRightClick = function(args)
		local self = args[2]
		local board = args[1]
		local player = board.players[board.currPlayer]
		if player.mana > 0 then
			player.mana = player.mana - 1
			self.changeAnimation("buff");
			self.Data.atk = self.Data.atk + 1
		end
	end
}
