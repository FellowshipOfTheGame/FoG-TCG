return{
	title = "Hikari",
	type = 'c',
	aspects = {0, 0, 0, 0},
	desc = "Devorar(+1)"
	flavor = "Não se sabe ao certo a origem dessa criatura. Com tantas similaridades com uma besta, deve mesmo ser tratada como humana?",
	cost = 3,
	atk = 2,
	hp = 3,

	buff = 0,

	OnOutgoingDamage = function(args)
		local self = args[2]
		args[4] = args[4] + self.Data.buff
	end,

	OnDamageDealt = function(args)
		local board, self, enemy, dmg = unpack(args)
		if enemy.Data.hp <= 0 then
			self.Data.buff = self.Data.buff + 1
		end
	end
}
