local attackBuff = function(args)
		args[4] = args[4] + 1
end

local resistOliphant = function(args)
		args[4] = args[4] - 1
end

return {
	title = "Carta 01",
	type = 'c',
	cost = 1,
	aspects = {0, 0, 0, 0},
	desc = "aparece na tela",
	flavor = "sem flavor",
	cost = 1,
	atk = 2,
	hp = 3,
	OnOutgoingDamage = attackBuff,
	OnIncomingDamage = resistOliphant,
	OnEnter = function(args)
		local self = args[1]
		self.Data.turnos_restantes = 3
	end,
	OnTurnEnd = function(args)
		local self = args[1]
		self.Data.turnos_restantes = self.Data.turnos_restantes - 1
		if self.Data.turnos_restantes == 0 then
			self.Remove()
		end
	end
}
