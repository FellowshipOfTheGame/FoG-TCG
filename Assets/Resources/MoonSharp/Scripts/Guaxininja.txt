return{
	title = "Guaxininja",
	type = 'c',
	aspects = {0, 0, 0, 0},
	desc = "Reposicionar",
	flavor = "Apos anos de missoes perigosas, esse guaxinim se aposentou e arranjou um emprego num restaurante chines, onde repassa seus ensinamentos de guerra por meio de biscoitinhos da sorte.",
	cost = 2,
	atk = 1,
	hp = 3,

	OnRightClick = function(args)
		local board = args[1]
		local self = args[2]
		
		board.WaitForTarget(self);
	end,

	OnChosenTarget = function(args)
		local board = args[1]
		local self = args[2]
		local lin = args[3]
		local col = args[4]

		local target = board.GetMatrixValue(lin, col);
		self.Move(lin, col);
		self.canCast = false
		self.canAttack = false
		if target ~= nil then
			target.canAttack = false
			target.canCast = false
		end
	end
}
